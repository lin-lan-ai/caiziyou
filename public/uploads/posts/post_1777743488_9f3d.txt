<script>
// === 后端接口（内嵌在HTML里，单文件即可运行） ===
const handler = async (req, res) => {
  // 全局存储（所有访问者共享）
  if (!globalThis.classmates) globalThis.classmates = [];

  const url = new URL(req.url, `http://${req.headers.host}`);
  const path = url.pathname;

  // 获取所有同学
  if (req.method === "GET" && path === "/list") {
    return res.end(JSON.stringify(globalThis.classmates));
  }

  // 添加同学
  if (req.method === "POST" && path === "/add") {
    let body = "";
    for await (const chunk of req) body += chunk;
    const data = JSON.parse(body);
    globalThis.classmates.push(data);
    res.end(JSON.stringify({ success: true }));
    return;
  }

  res.writeHead(404).end("Not Found");
};

// === 前端逻辑 ===
async function submit(){
  const name = document.getElementById("name").value.trim();
  const qq = document.getElementById("qq").value.trim();
  const msg = document.getElementById("msg").value.trim();
  if(!name) return alert("请填写姓名！");

  await fetch("/add",{
    method:"POST",
    headers:{"Content-Type":"application/json"},
    body:JSON.stringify({name,qq,msg})
  });
  alert("提交成功！");
  getList();
  document.getElementById("name").value = "";
  document.getElementById("qq").value = "";
  document.getElementById("msg").value = "";
}

async function getList(){
  const res = await fetch("/list");
  const data = await res.json();
  const list = document.getElementById("list");
  if(data.length===0){
    list.innerHTML = '<div class="empty">还没有人填写，快来第一个吧！</div>';
    return;
  }
  list.innerHTML = "";
  data.forEach(item=>{
    list.innerHTML += `
      <div class="card">
        <h3>${item.name}</h3>
        <p>QQ：${item.qq||"未填写"}</p>
        <p>留言：${item.msg||"无"}</p>
      </div>
    `;
  });
}

window.onload = getList;

// === Vercel 入口导出 ===
if (typeof module !== "undefined") module.exports = handler;
</script>
